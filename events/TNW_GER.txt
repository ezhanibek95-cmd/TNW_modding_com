country_event = {
    id = germany.1
    title = germany.1.t
    desc = germany.1.d
    
    trigger = {
        tag = GER
		date > 1952.1.1
    }
    
	mean_time_to_happen = {
		days = 150
	}
	fire_only_once = yes
    option = {
		name = germany.1.a  # Предположим, это вариант прихода Гиммлера
		add_stability = -0.10
		kill_country_leader = yes
		promote_character = GER_Heinrich_Himmler
		remove_advisor_role = {
			character = GER_Heinrich_Himmler
			slot = political_advisor
		}
		country_event = { id = germany.2 }  # новое событие для начала конфликта Геббельс–Гиммлер
	}
	option = {
		trigger = {
			power_balance_value = {
				id = GER_fascism_party_balance
				value < -0.20
			}
		}
        name = germany.1.b
        add_stability = -0.10
		kill_country_leader = yes
		promote_character = GER_Martin_Borman
		remove_advisor_role = {
			character = GER_Martin_Borman
			slot = political_advisor
		}
    }
	option = {
		trigger = {
			power_balance_value = {
				id = GER_fascism_party_balance
				value > -0.1
				value < 0.1
			}
		}
        name = germany.1.с
        add_stability = -0.10
		kill_country_leader = yes
		promote_character = GER_Joseph_Goebbels
		remove_advisor_role = {
			character = GER_Joseph_Goebbels
			slot = political_advisor
		}
    }
}
